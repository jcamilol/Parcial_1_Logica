\subsection*{\uline{Ejercicio1}}
Sea $\mathcal{E} = \{ \forall x \exists y P(x,y), \forall x \forall y (P(x,y) \rightarrow Q(y)), \forall x (Q(x) \rightarrow R(x)) \}$. Pruebe que $\mathcal{E} \vdash \forall x \exists y R(y)$.\\

\textbf{Solución:} Asumimos que $\mathcal{E}$ contiene los axiomas del lenguaje. Llamamos $E_1: \forall x \exists y P(x,y)$, $E_2: \forall x \forall y (P(x,y)\rightarrow Q(y))$ y $E_3: \forall x (Q(x) \rightarrow R(x))$. La siguiente es una deducción de $\forall x \exists y R(y)$ a partir de $\mathcal{E}$:

\begin{enumerate}
    \item $\forall x (Q(x) \to R(x))$ \hfill (Premisa $E_3$)
    \item $\forall x (Q(x) \to R(x)) \to (Q(y) \to R(y))$ \hfill (Axioma de especialización, $t=y$)
    \item $Q(y) \to R(y)$ \hfill (MP 1, 2)
    \item $\forall x \forall y (P(x,y) \to Q(y))$ \hfill (Premisa $E_2$)
    \item $\forall x \forall y (P(x,y) \to Q(y)) \to \forall y (P(x,y) \to Q(y))$ \hfill (Axioma de especialización, $t=x$)
    \item $\forall y (P(x,y) \to Q(y))$ \hfill (MP 4, 5)
    \item $\forall y (P(x,y) \to Q(y)) \to (P(x,y) \to Q(y))$ \hfill (Axioma de especialización, $t=y$)
    \item $P(x,y) \to Q(y)$ \hfill (MP 6, 7)
    \item $(P(x,y) \to Q(y)) \to ((Q(y) \to R(y)) \to (P(x,y) \to R(y)))$ \hfill (Tautología: Transitividad)
    \item $(Q(y) \to R(y)) \to (P(x,y) \to R(y))$ \hfill (MP 8, 9)
    \item $P(x,y) \to R(y)$ \hfill (MP 3, 10)
    \item $(P(x,y) \to R(y)) \to (\neg R(y) \to \neg P(x,y))$ \hfill (Tautología: Contraposición)
    \item $\neg R(y) \to \neg P(x,y)$ \hfill (MP 11, 12)
    \item $\forall y (\neg R(y) \to \neg P(x,y))$ \hfill (Generalización sobre $y$ en 13)
    \item $\forall y (\neg R(y) \to \neg P(x,y)) \to (\neg R(y) \to \forall y \neg P(x,y))$ \hfill (Axioma lógico para cuantificadores)
    \item $\neg R(y) \to \forall y \neg P(x,y)$ \hfill (MP 14, 15)
    \item $(\neg R(y) \to \forall y \neg P(x,y)) \to (\neg \forall y \neg P(x,y) \to \neg \neg R(y))$ \hfill (Tautología: Contraposición)
    \item $\neg \forall y \neg P(x,y) \to \neg \neg R(y)$ \hfill (MP 16, 17)
    \item $(\neg \forall y \neg P(x,y) \to \neg \neg R(y)) \to (\exists y P(x,y) \to R(y))$ \hfill (Axioma lógico para cuantificadores y eliminación de doble negación vía tautología)
    \item $\exists y P(x,y) \to R(y)$ \hfill (MP 18, 19)
    \item $\forall x \exists y P(x,y)$ \hfill (Premisa $E_1$)
    \item $\forall x \exists y P(x,y) \to \exists y P(x,y)$ \hfill (Axioma de especialización, $t=x$)
    \item $\exists y P(x,y)$ \hfill (MP 21, 22)
    \item $R(y)$ \hfill (MP 23, 20)
    \item $\forall y \neg R(y) \to \neg R(y)$ \hfill (Axioma de especialización, $t=y$)
    \item $(\forall y \neg R(y) \to \neg R(y)) \to (\neg \neg R(y) \to \neg \forall y \neg R(y))$ \hfill (Tautología: Contraposición)
    \item $\neg \neg R(y) \to \neg \forall y \neg R(y)$ \hfill (MP 25, 26)
    \item $R(y) \to \neg \neg R(y)$ \hfill (Tautología)
    \item $R(y) \to \neg \forall y \neg R(y)$ \hfill (Tautología: Transitividad 28, 27)
    \item $\neg \forall y \neg R(y)$ \hfill (MP 24, 29)
    \item $\exists y R(y)$ \quad \hfill (Axioma lógico para cuantificadores y eliminación de doble negación vía tautología)
    \item $\forall x \exists y R(y)$ \hfill (Generalización sobre $x$ en 30)
\end{enumerate}
\hfill \qed
